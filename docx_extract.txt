[434] from typing import AsyncGenerator
[436] class AIService:
[438] async def chat_stream(
[439] self,
[440] messages: list[dict],
[441] provider: str = None,
[442] model: str = None
[443] ) -> AsyncGenerator[str, None]:
[445] provider = provider or settings.AI_PROVIDER or 'openai'
[446] model = model or settings.AI_MODEL
[448] if provider == 'openai':
[449] client = openai.AsyncOpenAI(api_key=settings.OPENAI_API_KEY)
[450] model = model or 'gpt-4'
[451] stream = await client.chat.completions.create(
[452] model=model, messages=messages, stream=True, max_tokens=2000
[453] )
[454] async for chunk in stream:
[455] token = chunk.choices[0].delta.content
[456] if token: yield token
[458] elif provider == 'anthropic':
[459] client = anthropic.AsyncAnthropic(api_key=settings.ANTHROPIC_API_KEY)
[460] model = model or 'claude-3-sonnet-20240229'
[461] system = next((m['content'] for m in messages if m['role']=='system'), '')
[462] user_msgs = [m for m in messages if m['role'] != 'system']
[463] async with client.messages.stream(
[464] model=model, max_tokens=2000, system=system, messages=user_msgs
[465] ) as stream:
[466] async for text in stream.text_stream:
[467] yield text
[469] elif provider == 'google':
[470] genai.configure(api_key=settings.GOOGLE_API_KEY)
[471] model_name = model or 'gemini-pro'
[472] gmodel = genai.GenerativeModel(model_name)
[473] prompt = ' '.join(m['content'] for m in messages)
[474] response = await gmodel.generate_content_async(prompt, stream=True)
[475] async for chunk in response:
[476] if chunk.text: yield chunk.text
[478] elif provider == 'ollama':
[479] model = model or 'llama2'
[480] async with httpx.AsyncClient(timeout=120) as client:
[481] async with client.stream(
[482] 'POST',
[483] f'{settings.OLLAMA_BASE_URL}/api/chat',
[484] json={'model': model, 'messages': messages, 'stream': True}
[485] ) as resp:
[486] async for line in resp.aiter_lines():
[487] if line:
[488] import json
[489] data = json.loads(line)
[490] token = data.get('message', {}).get('content', '')
[491] if token: yield token
[493] else:
[494] yield f'Error: Unknown provider {provider}.'
[495] yield ' Please use: openai, anthropic, google, or ollama'
